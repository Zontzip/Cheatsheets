/***************************
	Simple Node Server
***************************/
-- Install simple http server
npm install http-server -g

-- Run server in current directory
http-server

/*********************
	Basic Commands
*********************/
-- Generate package.json based on Q's in current directory
npm init

-- Install a package and update package.json
npm install <module_name> --save

-- Update a package
npm update <module_name>

-- Uninstall a package
npm uninstall <module_name>

-- List everything installed in current directory
npm ls

-- Run test suite
npm test

-- List outdated libraries
npm outdated

-- Christmas tree
npm xmas

/*********************
	Project Setup
*********************/
-- Install all packages listed in package.json
npm install

-- Start webpack node server thing
npm start

/******************
	Permissions
******************/
-- Find NPM directory
npm config get prefix

-- Change NPM directory owner (presumed to be lib/node_modules)
sudo chown -R $(whoami) $(npm config get prefix)/{lib/node_modules,bin,share}
